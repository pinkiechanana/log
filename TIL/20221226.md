```
<a href="/create" onClick={(event)=>{
        event.preventDefault();  // url 바뀌는거 방지
        setMode('CREATE');       // mode의 state를 변경
      }}>Create</a>

function App () {
  const [mode, setMode] = useState('Welcome');
  const [id, setId] = useState(null);
  const [nextId, setNextId] = useState(4);
  const [topics, setTopics] = useState([
    {id:1, title:'Html', body:'html example'},
    {id:2, title:'CSS', body:'css example'},
    {id:3, title:'Javascript', body:'javascript example'},
  ]); // array, object도 가능. 대신 state update 할때 spread operator 이용해야됨

  let content = null;
  if (mode === 'WELCOME') {
    // ...
  } else if (mode === 'READ') {
    // ...
  } else if (mode === 'CREATE') {
    content = 
    <Create onCreate={(_title, _body)=>{
      const newTopic = {id: nextId, title:_title, body:_body}
      const newTopics = [...topics]    // Spread operator 이용(그 전 state를 
                                       // overwrite하데 전 데이터도 같이 적용되게)
                                       // 차나나 이 과정 다시 보셈!!!!!!!!!!!
      newTopics.push(newTopic);
      setTopics(newTopics);
    }}></Create>
  }
}

function Create(props) {
  return 
  <article>
    <form onSubmit={event=>{  // onSubmit: submit 버튼이 눌려졌을때 form tag에서 발생하는 event
      event.preventDefault();
      const title = event.target.title.value; // targets on the form itself
      const body = event.target.body.value;   // targets on the form itself
      props.onCreate(title, body);   // Line:45
    }}>
      <p><input type="text" name="title" placeholder="Title"/></p>
      <p><textarea name="body" placeholder="Body"></textarea></p>
      <p><input type="submit" value="Create"></input></p>
    </form>
    <h2>Create</h2>
  </article>
}
```
